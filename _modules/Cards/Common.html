[docs]def std_dev(input_list=[], mean=None ):
	if(type(mean)==type(None) ):
		mean=average(input_list)
	return sum( [abs( (float(x)-mean)*(float(x)-mean) ) for x in input_list  ] )/float(len( input_list))

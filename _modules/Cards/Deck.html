class _Card_List:
	def __init__(self):

	def __str__(self):
		tmp_string = ""
		for i in self.cards:
			tmp_string+="\t"+str(i)+"\n"
		return tmp_string

[docs]	def get_card_before(self, card):
		if(not type(card) == type(Card.Card() )):
			TypeError("Expected input of Type Card. Received: "+str(type(card) ) )
		before_card = Card.Card(0,0)
		for i in self.private.cards :
			if (i == card ):
				return before_card
			before_card = i
		return Card.Card(0,0)

		if(not type(card) == type(Card.Card() )):
			TypeError("Expected input of Type Card. Received: "+str(type(card) ) )
		found_card = False
		for i in self.private.cards :
			if(found_card):
				return i
			if(i==card):
				found_card = True
		return Card.Card(0,0)
